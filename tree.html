render() {
                stroke(0, 175);
                for (let i = 0; i < this.todo.length; i++) {
                    let c = this.todo[i];
                    if (c == 'F') {
                        line(0, 0, this.len, 0);
                        translate(this.len, 0);
                    } else if (c == 'G') {
                        translate(this.len, 0);
                    } else if (c == '+') {
                        rotate(this.theta);
                    } else if (c == '-') {
                        rotate(-this.theta);
                    } else if (c == '[') {
                        push();
                    } else if (c == ']') {
                        pop();
                    }
                }
            }

            new Rule('F', "FF+[+F-F-F]-[-F+F+F]")
